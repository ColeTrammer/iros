18 constexpr Out operator()(It first, Sent last, Out out, meta::TypeIdentity<SSizeType> n, Gen&& generator) const {
19 using Distribution = random::UniformIntDistribution<SSizeType>;
20 using Param = Distribution::Param;
21
22 auto distribution = Distribution {};
23
24 if constexpr (concepts::ForwardIterator<It>) {
25 // Implement selection sampling.
26
27 auto count = container::distance(first, last);
28 for (auto to_sample = container::min(count, n); to_sample != 0; --first) {
29 // Generate a random number between [0, count).
30 // Only keep the current element if the random number
31 // chosen is less than the number of samples which have
32 // not yet been generated.
33 if (distribution(generator, Param(0, --count)) < to_sample) {
34 *out = *first;
35 ++out;
36 --to_sample;
37 }
38 }
39 return out;
40 } else {
41 // Implement reservoir sampling, storing extra elements in the output iterator's storage.
42 // This is Algorithm R, as described on wikipedia: https://en.wikipedia.org/wiki/Reservoir_sampling.
43
44 // Copy the first n elements into the reservoir. If the input size is less than n,
45 // we will have no more work to do.
46 SSizeType reservoir_size = 0;
47 for (; first != last && reservoir_size < n; ++first, ++reservoir_size) {
48 out[reservoir_size] = *first;
49 }
50
51 // Overwrite the reservoir with new elements if they get selected by the algorithm.
52 for (auto size_so_far = reservoir_size; first != last; ++first, ++size_so_far) {
53 auto chosen_index = distribution(generator, Param(0, size_so_far));
54 if (chosen_index < n) {
55 out[chosen_index] = *first;
56 }
57 }
58
59 return out;
60 }
61 }
