288 template<concepts::Encoding Enc>
289 constexpr Result<void>
290 present_string_view_to(concepts::FormatContext auto& context, Optional<FillAndAlign> fill_and_align,
291 Optional<size_t> width, Optional<size_t> precision, bool debug,
292 container::string::StringViewImpl<Enc> view_in, char32_t delimit_code_point = U'"') {
293 using CodePoint = meta::EncodingCodePoint<Enc>;
294
295 auto delimit = lift_bool(debug) % function::value(delimit_code_point);
296 auto view = view::concat(delimit, view_in, delimit);
297
298 auto measure_code_point = [&](CodePoint) -> size_t {
299 (void) debug;
300 return 1;
301 };
302
303 size_t width_printed_so_far = 0;
304 auto output_char = [&](CodePoint code_point) -> Result<void> {
305 // Fast path when there is no precision.
306 if (!precision) {
307 context.output(code_point);
308 return {};
309 }
310
311 // Don't print characters after the precision is exceeded.
312 auto code_point_width = measure_code_point(code_point);
313 if (width_printed_so_far + code_point_width > *precision) {
314 return {};
315 }
316
317 context.output(code_point);
318 width_printed_so_far += code_point_width;
319 return {};
320 };
321
322 if (!width) {
323 return container::sequence(view, output_char);
324 }
325
326 auto total_width = view | view::transform(measure_code_point) | container::sum;
327 if (total_width >= *width) {
328 return container::sequence(view, output_char);
329 }
330
331 auto align = fill_and_align.transform(&FillAndAlign::align).value_or(FillAndAlign::Align::Left);
332 auto fill_code_point = fill_and_align.transform(&FillAndAlign::fill).value_or(U' ');
333
334 auto chars_to_pad = (*width - total_width) / measure_code_point(fill_code_point);
335 auto [left_pad, right_pad] = [&]() -> Tuple<size_t, size_t> {
336 switch (align) {
337 case FillAndAlign::Align::Left:
338 return { 0, chars_to_pad };
339 case FillAndAlign::Align::Center:
340 return { chars_to_pad / 2, math::divide_round_up(chars_to_pad, 2u) };
341 case FillAndAlign::Align::Right:
342 return { chars_to_pad, 0 };
343 default:
344 util::unreachable();
345 }
346 }();
347
348 auto do_pad = [&](auto) {
349 return output_char(fill_code_point);
350 };
351
352 DI_TRY(container::sequence(view::range(left_pad), do_pad));
353 DI_TRY(container::sequence(view, output_char));
354 return container::sequence(view::range(right_pad), do_pad);
355 }

369 template<concepts::Encoding Enc, concepts::Integral T>
370 constexpr Result<void>
371 present_integer_to(concepts::FormatContext auto& context, Optional<FillAndAlign> fill_and_align, Sign sign,
372 HashTag hash_tag, Zero zero, Optional<size_t> width, IntegerType type, bool debug, T value) {
373 if (type == IntegerType::Character) {
374 return present_character_to<Enc>(context, fill_and_align, width, debug, static_cast<c32>(value));
375 }
376
377 using CodePoint = meta::EncodingCodePoint<Enc>;
378
379 // The maximum number of digits a number can have is 64 (u64::max() printed in binary).
380 // Add 3 extra characters to account for a prefix, like -0x.
381 auto buffer = container::string::StringImpl<
382 Enc, container::StaticVector<meta::EncodingCodeUnit<Enc>, meta::Constexpr<67zu>>> {};
383
384 using UnsignedType = meta::MakeUnsigned<T>;
385 auto as_unsigned = math::abs_unsigned(value);
386
387 auto const negative = [&] {
388 if constexpr (concepts::Signed<T>) {
389 return value < 0;
390 }
391 return false;
392 }();
393
394 auto do_sign = [&] {
395 switch (sign) {
396 case Sign::Minus:
397 if (negative) {
398 (void) buffer.push_back(CodePoint('-'));
399 }
400 break;
401 case Sign::Plus:
402 if (negative) {
403 (void) buffer.push_back(CodePoint('-'));
404 } else {
405 (void) buffer.push_back(CodePoint('+'));
406 }
407 break;
408 case Sign::Space:
409 if (negative) {
410 (void) buffer.push_back(CodePoint('-'));
411 } else {
412 (void) buffer.push_back(CodePoint(' '));
413 }
414 break;
415 }
416 };
417
418 auto do_prefix = [&] {
419 if (hash_tag == HashTag::No) {
420 return;
421 }
422 switch (type) {
423 case IntegerType::BinaryLower:
424 (void) buffer.push_back(CodePoint('0'));
425 (void) buffer.push_back(CodePoint('b'));
426 break;
427 case IntegerType::BinaryUpper:
428 (void) buffer.push_back(CodePoint('0'));
429 (void) buffer.push_back(CodePoint('B'));
430 break;
431 case IntegerType::Octal:
432 (void) buffer.push_back(CodePoint('0'));
433 break;
434 case IntegerType::HexLower:
435 (void) buffer.push_back(CodePoint('0'));
436 (void) buffer.push_back(CodePoint('x'));
437 break;
438 case IntegerType::HexUpper:
439 (void) buffer.push_back(CodePoint('0'));
440 (void) buffer.push_back(CodePoint('X'));
441 break;
442 default:
443 break;
444 }
445 };
446
447 bool zero_pad = zero == Zero::Yes && !fill_and_align;
448
449 do_sign();
450 do_prefix();
451
452 if (zero_pad) {
453 for (auto ch : buffer) {
454 context.output(ch);
455 }
456 auto code_points = math::to_unsigned(container::distance(buffer));
457 if (width && *width > code_points) {
458 *width -= code_points;
459 }
460 buffer.clear();
461 }
462
463 auto const radix = [&] -> UnsignedType {
464 switch (type) {
465 case IntegerType::BinaryLower:
466 case IntegerType::BinaryUpper:
467 return 2;
468 case IntegerType::Octal:
469 return 8;
470 case IntegerType::HexLower:
471 case IntegerType::HexUpper:
472 return 16;
473 default:
474 return 10;
475 }
476 }();
477
478 auto to_digit = [&](UnsignedType value) -> meta::EncodingCodePoint<Enc> {
479 switch (type) {
480 case IntegerType::HexLower:
481 return (value >= 10) ? ('a' + (value - 10)) : ('0' + value);
482 case IntegerType::HexUpper:
483 return (value >= 10) ? ('A' + (value - 10)) : ('0' + value);
484 default:
485 return ('0' + value);
486 }
487 };
488
489 UnsignedType strength = 1;
490 for (auto x = as_unsigned; x / strength >= radix; strength *= radix) {}
491
492 for (; strength; strength /= radix) {
493 (void) buffer.push_back(to_digit((as_unsigned / strength) % radix));
494 }
495
496 auto backup_fill_and_align = FillAndAlign { zero_pad ? U'0' : U' ', FillAndAlign::Align::Right };
497 return present_string_view_to<Enc>(context, fill_and_align.value_or(backup_fill_and_align), width, nullopt,
498 false, buffer);
499 }

501 template<concepts::Encoding Enc>
502 constexpr Result<void> present_formatted_to(concepts::FormatContext auto& context,
503 Optional<FillAndAlign> fill_and_align, Optional<size_t> width,
504 Optional<size_t> precision,
505 container::string::StringViewImpl<Enc> format_string, auto&&... args) {
506 // If there is no width, fill_and_align is ignored, so no temporary buffer is needed.
507
508 // Precision in this case refers to the upper bound on the text width to be printed.
509 // Handle precision using a modified context which caps to number of characters printed,
510 // in the case where there is no width.
511
512 // If width is present, we must first collect the input into a temporary buffer before presenting.
513 // Performing 2 passes is not viable, as we may be printing InputContainers, which are only valid
514 // for a single pass, and anyway, iterating over a large vector twice should be avoided.
515
516 // The temporary buffer should include a sizable inline capacity (~256 bytes), to prevent heap
517 // allocation except for in extreme cases. Note that if the width is smaller than 256, formatting
518 // can be successful without ever allocating. However, the temporary buffer should be customizable,
519 // so that in kernel context, allocation can never happen (and perhaps less stack size should be
520 // used, since the kernel may only have 4096 bytes of stack space).
521
522 DI_ASSERT(!width);
523 DI_ASSERT(!precision);
524 (void) fill_and_align;
525
526 return vpresent_encoded_context<Enc>(context, format_string, make_constexpr_format_args(args...));
527 }
