20template<typename T, typename U>
21void binary_assert_fail(char const* message, T&& a, U&& b, util::SourceLocation loc) {
22 // Allocate a 256 byte buffer on the stack to stringify a and b. We'd don't want to allocate here because this
23 // assertion could indicate heap corruption, or even be triggered before the heap is initialized.
24 using Enc = container::string::Utf8Encoding;
25 using TargetContext = format::BoundedFormatContext<Enc, meta::Constexpr<256zu>>;
26
27 auto a_context = TargetContext {};
28 auto const* a_data_pointer = static_cast<char const*>(nullptr);
29 if constexpr (concepts::Formattable<T>) {
30 (void) di::format::vpresent_encoded_context<Enc>(
31 di::container::string::StringViewImpl<Enc>(encoding::assume_valid, u8"{}", 2),
32 di::format::make_format_args<TargetContext>(a), a_context);
33 a_data_pointer = reinterpret_cast<char const*>(a_context.output().span().data());
34 }
35 auto b_context = TargetContext {};
36 auto const* b_data_pointer = static_cast<char const*>(nullptr);
37 if constexpr (concepts::Formattable<U>) {
38 (void) di::format::vpresent_encoded_context<Enc>(
39 di::container::string::StringViewImpl<Enc>(encoding::assume_valid, u8"{}", 2),
40 di::format::make_format_args<TargetContext>(b), b_context);
41 b_data_pointer = reinterpret_cast<char const*>(b_context.output().span().data());
42 }
43 assert_fail(message, a_data_pointer, b_data_pointer, loc);
44}
