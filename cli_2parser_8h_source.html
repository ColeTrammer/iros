46 constexpr Result<Base> parse(Span<TransparentStringView> args) {
47 using namespace di::string_literals;
48
49 if (args.empty()) {
50 return Unexpected(BasicError::InvalidArgument);
51 }
52 args = *args.subspan(1);
53
54 auto seen_arguments = Array<bool, sizeof...(Options)> {};
55 seen_arguments.fill(false);
56
57 auto count_option_processed = usize { 0 };
58
59 auto send_arg_to_back = [&](usize i) {
60 container::rotate(*args.subspan(i), args.begin() + i + 1);
61 count_option_processed++;
62 };
63
64 auto result = Base {};
65 for (usize i = 0; i < args.size(); i++) {
66 auto arg_index = i - count_option_processed;
67 auto arg = args[arg_index];
68
69 // Handle positional argument.
70 if (!arg.starts_with('-')) {
71 continue;
72 }
73
74 // Handle exactly "--".
75 if (arg == "--"_tsv) {
76 send_arg_to_back(arg_index);
77 break;
78 }
79
80 // Handle short argument.
81 if (!arg.starts_with("--"_tsv)) {
82 for (usize char_index = 1; char_index < arg.size(); char_index++) {
83 auto index = lookup_short_name(arg[char_index]);
84 if (!index) {
85 return Unexpected(BasicError::InvalidArgument);
86 }
87
88 // Parse boolean flag.
89 if (option_boolean(*index)) {
90 DI_TRY(option_parse(*index, seen_arguments, &result, {}));
91 continue;
92 }
93
94 // Parse short option with directly specified value: '-iinput.txt'
95 auto value_view = arg.substr(arg.begin() + char_index + 1);
96 if (!value_view.empty()) {
97 DI_TRY(option_parse(*index, seen_arguments, &result, value_view));
98 break;
99 }
100
101 // Fail if the is no subsequent arguments left.
102 if (i + 1 >= args.size()) {
103 return Unexpected(BasicError::InvalidArgument);
104 }
105
106 // Use the next argument as the value.
107 DI_TRY(option_parse(*index, seen_arguments, &result, args[arg_index + 1]));
108 send_arg_to_back(arg_index);
109 i++;
110 }
111 send_arg_to_back(arg_index);
112 continue;
113 }
114
115 // Handle long arguments.
116 auto equal = arg.find('=');
117 auto name = ""_tsv;
118 if (!equal) {
119 name = arg.substr(arg.begin() + 2);
120 } else {
121 name = arg.substr(arg.begin() + 2, equal.begin());
122 }
123
124 auto index = lookup_long_name(name);
125 if (!index) {
126 return Unexpected(BasicError::InvalidArgument);
127 }
128
129 if (option_boolean(*index)) {
130 if (equal) {
131 return Unexpected(BasicError::InvalidArgument);
132 }
133 DI_TRY(option_parse(*index, seen_arguments, &result, {}));
134 send_arg_to_back(arg_index);
135 continue;
136 }
137
138 auto value = ""_tsv;
139 if (!equal && i + 1 >= args.size()) {
140 return Unexpected(BasicError::InvalidArgument);
141 } else if (!equal) {
142 value = args[arg_index + 1];
143 send_arg_to_back(arg_index);
144 send_arg_to_back(arg_index);
145 i++;
146 } else {
147 value = arg.substr(equal.end());
148 send_arg_to_back(arg_index);
149 }
150
151 DI_TRY(option_parse(*index, seen_arguments, &result, value));
152 }
153
154 // Validate all required arguments were processed.
155 for (usize i = 0; i < sizeof...(Options); i++) {
156 if (!seen_arguments[i] && option_required(i)) {
157 return Unexpected(BasicError::InvalidArgument);
158 }
159 }
160
161 // All the positional arguments are now at the front of the array.
162 auto positional_arguments = *args.subspan(0, args.size() - count_option_processed);
163 if (positional_arguments.size() < minimum_required_argument_count()) {
164 return Unexpected(BasicError::InvalidArgument);
165 }
166
167 auto argument_index = usize(0);
168 for (auto i = usize(0); i < positional_arguments.size(); argument_index++) {
169 auto count_to_consume = !argument_variadic(i) ? 1 : positional_arguments.size() - argument_count() + 1;
170 auto input = *positional_arguments.subspan(i, count_to_consume);
171 DI_TRY(argument_parse(argument_index, &result, input));
172 i += count_to_consume;
173 }
174 return result;
175 }
