149 void report_value(Constexpr<index>, Types&&... values) {
150 if constexpr (never_sends_value<Env, Sends...>) {
151 ((void) values, ...);
152 } else {
153 auto& optional = util::get<index>(value);
154 optional.emplace(util::forward<Types>(values)...);
155 }
156 finish_one();

159 template<typename E>
160 void report_error(E&& error) {
161 auto old = failed.exchange(true, sync::MemoryOrder::AcquireRelease);
162 if (!old) {
163 stop_source.request_stop();
164 this->error.template emplace<meta::Decay<E>>(util::forward<E>(error));
165 }
166 finish_one();
167 }
168
169 void report_stop() {
170 auto old = failed.exchange(true, sync::MemoryOrder::AcquireRelease);
171 if (!old) {
172 stop_source.request_stop();
173 error.template emplace<Stopped>();
174 }
175 finish_one();
176 }
177
178 void finish_one() {
179 auto old_value = remaining.fetch_sub(1, sync::MemoryOrder::AcquireRelease);
180 if (old_value == 1) {
181 // Reset the stop callback.
182 stop_callback.reset();
183
184 // Check if an error occurred. This does not need to be atomic because we're done and just used
185 // acquire release ordering.
186 if (!vocab::holds_alternative<NotError>(error)) {
187 if (!vocab::holds_alternative<Stopped>(error)) {
188 vocab::visit(
189 [&]<typename E>(E&& e) {
190 if constexpr (!concepts::OneOf<meta::RemoveCVRef<E>, NotError, Stopped>) {
191 execution::set_error(util::move(out_r), util::forward<E>(e));
192 }
193 },
194 util::move(error));
195 } else {
196 execution::set_stopped(util::move(out_r));
197 }
198 } else {
199 if constexpr (!never_sends_value<Env, Sends...>) {
200 // Value is a Tuple<Optional<Vs...>, ...>, and we need to return all Vs's. Do this be
201 // concatenating all the tuples and then unpacking them.
202 auto lvalues = vocab::apply(
203 [](auto&... optionals) {
204 return vocab::tuple_cat(vocab::apply(vocab::tie, *optionals)...);
205 },
206 value);
207 vocab::apply(
208 [&](auto&... values) {
209 execution::set_value(util::move(out_r), util::move(values)...);
210 },
211 lvalues);
212 } else {
213 util::unreachable();
214 }
215 }
216 }
217 }

220 [[no_unique_address]] Error error;
221 [[no_unique_address]] Rec out_r;
222 sync::InPlaceStopSource stop_source;

224 sync::Atomic<bool> failed { false };
