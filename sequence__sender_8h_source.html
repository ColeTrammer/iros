201 struct Function {
202 template<concepts::Receiver Rec, concepts::SenderIn<meta::EnvOf<Rec>> Seq>
203 requires(concepts::detail::AdaptableToSequence<Rec, Seq> ||
204 (concepts::SubscriberFrom<Rec, Seq> && concepts::TagInvocable<Function, Seq, Rec>) ||
205 (concepts::DebugEnv<meta::EnvOf<Rec>>) )
206 auto operator()(Seq&& sequence, Rec&& receiver) const {
207 if constexpr (concepts::detail::AdaptableToSequence<Rec, Seq>) {
208 return connect(
209 set_next(receiver, util::forward<Seq>(sequence)),
210 execution::sender_to_sequence_adaptor_ns::Receiver<Rec> { util::forward<Rec>(receiver) });
211 } else if constexpr (concepts::SubscriberFrom<Rec, Seq> && concepts::TagInvocable<Function, Seq, Rec>) {
212 static_assert(
213 concepts::OperationState<meta::TagInvokeResult<Function, Seq, Rec>>,
214 "The return value of di::execution::subscribe() must model di::concepts::OperationState.");
215 return function::tag_invoke(*this, util::forward<Seq>(sequence), util::forward<Rec>(receiver));
216 } else if constexpr (!concepts::SequenceSender<Seq>) {
217 return connect(
218 set_next(receiver, util::forward<Seq>(sequence)),
219 execution::sender_to_sequence_adaptor_ns::Receiver<Rec> { util::forward<Rec>(receiver) });
220 } else {
221 return function::tag_invoke(*this, util::forward<Seq>(sequence), util::forward<Rec>(receiver));
222 }
223 }
224 };
