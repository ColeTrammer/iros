142 static OperationState<Receiver> impl(Awaitable awaitable, Receiver receiver) {
143 using Result = meta::AwaitResult<Awaitable, Promise<Receiver>>;
144
145 // Connecting any awaitable with a receiver is a matter of returning an operation
146 // state, which, once started, enters coroutine context, calls co_await on the awaitable,
147 // suspends said coroutine, and finally calls the receiver's completion with the reuslt
148 // of co_await.
149
150 // To do so, the OperationState is a coroutine (has a promise type), whose start operation
151 // resumes the coroutine. To forcefully suspend the coroutine, we co_yield a lambda expression,
152 // which both suspends the coroutine, but also executes the lambda once the suspension occurs.
153 if constexpr (concepts::LanguageVoid<Result>) {
154 co_await util::move(awaitable);
155 co_yield [&] {
156 set_value(util::move(receiver));
157 };
158 } else {
159 auto&& value = co_await util::move(awaitable);
160 co_yield [&] {
161 set_value(util::move(receiver), util::forward<decltype(value)>(value));
162 };
163 }
164 }
