563inline void IoUringContext::run() {
564 for (;;) {
565 // Reap any pending completions.
566 while (auto cqe = m_handle.get_next_cqe()) {
567 auto* as_operation = reinterpret_cast<OperationStateBase*>(cqe->user_data);
568 as_operation->did_complete(cqe.data());
569 }
570
571 // Run locally available operations.
572 while (!m_queue.empty()) {
573 auto& item = *m_queue.pop();
574 item.execute();
575 }
576
577 // If we're done, we're done.
578 if (m_done) {
579 break;
580 }
581
582 // Wait for some event to happen.
583 (void) m_handle.submit_and_wait();
584 }

588inline void enqueue_io_operation(IoUringContext* context, OperationStateBase* op, Fun&& function) {
589 auto sqe = context->m_handle.get_next_sqe();
590 ASSERT(sqe);
591 di::fill_n(reinterpret_cast<di::Byte*>(sqe.data()), sizeof(sqe), 0_b);
592 di::invoke(di::forward<Fun>(function), sqe.data());
593 sqe->user_data = reinterpret_cast<uintptr_t>(op);
594}
