40 template<typename T>
41 struct ObjectWithRefCount : util::Immovable {
42 public:
43 ObjectWithRefCount(ObjectWithRefCount const&) = delete;
44 ObjectWithRefCount(ObjectWithRefCount&&) = delete;
45
46 template<typename... Args>
47 constexpr ObjectWithRefCount(Args&&... args) : ref_count(1), object(util::forward<Args>(args)...) {}
48
49 // Shared storage internally stores a pointer directly to the underlying T object. To access the reference
50 // count, shared storage looks directly in front of the object. To ensure this layout is chosen by the compiler,
51 // we must pad this object if the underlying T is over-aligned.
52 constexpr static usize alignment = container::max(alignof(T), alignof(sync::Atomic<usize>));
53 constexpr static usize padding_size =
54 alignment <= alignof(sync::Atomic<usize>) ? 0 : alignment - sizeof(sync::Atomic<usize>);
55
56 static ObjectWithRefCount* from_object_pointer(T* object) {
57 auto* byte_pointer = reinterpret_cast<byte*>(object);
58 byte_pointer -= padding_size + sizeof(sync::Atomic<usize>);
59 return reinterpret_cast<ObjectWithRefCount*>(byte_pointer);
60 }
61
62 T* to_object_pointer() { return util::addressof(object); }
63
64 [[no_unique_address]] Array<byte, padding_size> padding;
65 sync::Atomic<usize> ref_count;
66 T object;
67 };
