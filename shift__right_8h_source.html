13 struct ShiftRightFunction {
14 template<concepts::Permutable It, concepts::SentinelFor<It> Sent>
15 constexpr View<It> operator()(It first, Sent last, meta::IteratorSSizeType<It> n) const {
16 DI_ASSERT(n >= 0);
17
18 if (n == 0) {
19 return { first, container::next(first, last) };
20 }
21
22 if constexpr (concepts::BidirectionalIterator<It>) {
23 auto last_it = container::next(first, last);
24 auto new_start = last_it;
25 container::advance(new_start, -n, first);
26
27 if (new_start == first) {
28 return { last_it, last_it };
29 }
30
31 auto real_start = container::move_backward(first, new_start, last_it).out;
32 return { real_start, last_it };
33 } else {
34 auto new_start = first;
35 container::advance(new_start, n, last);
36 if (new_start == last) {
37 return { new_start, new_start };
38 }
39
40 // Use fast and slow pointers, with the fast pointer
41 // n elements ahead of the slow pointer. This approach
42 // is outlined in the C++ paper for shift_left and shift_right.
43 // See https://www.open-std.org/jtc1/sc22/wg21/docs/papers/2017/p0769r0.pdf.
44 auto fast = new_start;
45 auto slow = first;
46
47 // Fast path: the shift window is such that the size of the container is
48 // less than twice the shift amount. This lets us do the shift directly,
49 // without moving things around as we go.
50 for (; slow != new_start; ++fast, ++slow) {
51 if (fast == last) {
52 auto result = container::move(first, slow, new_start);
53 return { new_start, result.out };
54 }
55 }
56
57 // Cyclic path, use the first n elements of the input as storage
58 // for the elements which must be shifted into place. This approach essentially
59 // works on blocks of size n. At each block, we swap the elements at the start
60 // of the input with the elements of the current block. This makes the current
61 // block point to the correct values, which retaining the pending elements which
62 // otherwise would be overridden. Once we hit the end of the container, perform
63 // a final move operation to finish the shift.
64 for (;;) {
65 for (auto pending = first; pending != new_start; ++fast, ++slow, ++pending) {
66 if (fast == last) {
67 // Move the pending elements into place.
68 slow = container::move(pending, new_start, slow).out;
69
70 // Move the final elements into place.
71 auto last_it = container::move(first, pending, slow).out;
72 return { new_start, last_it };
73 }
74 // Swap the contents of slow and pending. The pending
75 // element of the previous iteration is now in the correct
76 // place, while pending now holds an element which will be
77 // shifted in later.
78 container::iterator_swap(pending, slow);
79 }
80 }
81 }
82 }
