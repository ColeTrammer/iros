223 struct ErasedStorage {
224 union {
225 void* pointer;
226 alignas(2 * sizeof(void*)) di::Array<di::Byte, 2 * sizeof(void*)> byte_storage;
227 };
228
229 void* address() { return util::addressof(byte_storage[0]); }
230 void const* address() const { return util::addressof(byte_storage[0]); }

233 template<typename F>
234 concept StoredInline =
235 sizeof(F) <= sizeof(ErasedStorage) && alignof(F) <= alignof(ErasedStorage) && concepts::MoveConstructible<F>;
236
237 template<concepts::Allocator Alloc>
238 struct ErasedObject {
239 private:
240 using ThunkFunction = void (*)(ErasedStorage*, ErasedStorage*, Alloc&);
241
242 template<typename T>
243 static void concrete_thunk(ErasedStorage* a, ErasedStorage* b, Alloc& allocator) {
244 if constexpr (StoredInline<T>) {
245 if (b) {
246 // Move from b into a.
247 auto* b_value = static_cast<T*>(b->address());
248 util::construct_at(static_cast<T*>(a->address()), util::move(*b_value));
249 util::destroy_at(b_value);
250 } else {
251 // Just destroy a.
252 util::destroy_at(static_cast<T*>(a->address()));
253 }
254 } else {
255 if (b) {
256 // Move from b into a.
257 a->pointer = util::exchange(b->pointer, nullptr);
258 } else {
259 // Just destroy a.
260 auto* a_value = static_cast<T*>(a->pointer);
261 util::destroy_at(a_value);
262 di::deallocate_one<T>(allocator, a_value);
263 }
264 }
265 }
266
267 public:
268 ErasedObject() {}
269
270 ErasedObject(ErasedObject const&) = delete;
271 ErasedObject(ErasedObject&& other) {
272 if ((m_thunk = util::exchange(other.m_thunk, nullptr))) {
273 m_thunk(util::addressof(m_storage), util::addressof(other.m_storage), m_allocator);
274 }
275 }
276
277 ~ErasedObject() { reset(); }
278
279 ErasedObject& operator=(ErasedObject const&) = delete;
280 ErasedObject& operator=(ErasedObject&& other) {
281 reset();
282 if ((m_thunk = util::exchange(other.m_thunk, nullptr))) {
283 m_thunk(util::addressof(m_storage), util::addressof(other.m_storage), m_allocator);
284 }
285 return *this;
286 }

288 bool empty() const { return m_thunk != nullptr; }
289 void reset() {
290 if (auto* old_thunk = util::exchange(m_thunk, nullptr)) {
291 old_thunk(util::addressof(m_storage), nullptr, m_allocator);
292 }
