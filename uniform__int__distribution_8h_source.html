44 constexpr T operator()(Gen&& generator, Param const& param) const {
45 using U = meta::CommonType<meta::MakeUnsigned<T>, typename meta::RemoveReference<Gen>::Result>;
46
47 // TODO: adopt a more sophisticated approach for implementing this function
48 // which does not naively retry the number generation on out of bounds.
49 // See 2018 paper: Fast Random Integer Generation in an Interval
50 // DANIEL LEMIRE, Université du Québec (TELUQ), Canada
51 constexpr U generated_min = meta::RemoveReference<Gen>::min();
52 constexpr U generated_max = meta::RemoveReference<Gen>::max();
53 constexpr U generated_range = generated_max - generated_min;
54
55 // A general simplification of generating an integer in the range
56 // [a, b] is to consider generating an integer from [0, b - a] instead,
57 // making sure to add a back in the end. When sampling from the generating
58 // range, all that matters is the size of its output range, and so generated
59 // values are always normalized by subtracting the generator's minimum value.
60 auto generate = [&] -> U {
61 return generator() - generated_min;
62 };
63
64 U a = param.a();
65 U b = param.b();
66 U desired_range = b - a;
67
68 if (generated_range == desired_range) {
69 // Simply sample the generated range, since it has the same
70 // size as the desired range.
71 return a + generate();
72 }
73
74 if (generated_range > desired_range) {
75 // Down sample the generated bits.
76 // Imagine having a generated range [0, 15], and a desired range [0, 2].
77 // To down sample the generated range, note that the desired range has 3
78 // elements, and the generated range has 16. Therefore, generate a random value
79 // and discard it if it has the value 15. Otherwise, dividing by 5 will give
80 // a properly sampled integer.
81 U const desired_size = desired_range + 1;
82 U const scale_factor = generated_range / desired_size;
83 U const out_of_bounds = desired_size * scale_factor;
84 for (;;) {
85 auto result = generate();
86 if (result >= out_of_bounds) {
87 continue;
88 }
89 return a + result / scale_factor;
90 }
91 } else {
92 // Up sample the generated bits.
93 // Imagine having a generated range [0, 3], and a desired range [0, 12].
94 // To up sample the generated range, realize that the 13 element space in
95 // the desired range can be partitioned into 4 parts, one for each possible
96 // outcome of the generated range. So, recusively generate a uniform integer in
97 // the range [0, 3] and multiply that by 4 to get an index in { 0, 4, 8, 12 }.
98 // Then add to this index a non-scaled generated value, to produce a uniform
99 // integer in the range [0, 15]. And finally, discard the value if it is out of bounds.
100 U const generated_size = U(generated_range + 1);
101 for (;;) {
102 U const base_index = generated_size * (*this)(generator, Param(0u, desired_range / generated_size));
103 U const index = generate();
104 U const result = base_index + index;
105
106 // Try again result is out of bounds (this also checks if the computation overflowed).
107 if (result > desired_range || result < base_index) {
108 continue;
109 }
110 return a + result;
111 }
112 }
113 }
