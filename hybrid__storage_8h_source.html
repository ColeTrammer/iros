166 template<typename T>
167 void HybridStorageManage<HybridStorage, Alloc>::operator()(T& a, HybridStorage* as, HybridStorage* b,
168 Alloc& allocator) const {
169 if constexpr (!HybridStorage::creation_is_inline(in_place_type<T>)) {
170 if (b) {
171 if constexpr (HybridStorage::storage_category() == StorageCategory::MoveOnly) {
172 // Move from b into a.
173 as->m_pointer = util::exchange(b->m_pointer, nullptr);
174 } else {
175 DI_ASSERT(!b);
176 }
177 } else {
178 // Just destroy a.
179 auto* pointer = util::exchange(as->m_pointer, nullptr);
180 auto* a_value = static_cast<T*>(pointer);
181 util::destroy_at(a_value);
182 di::deallocate_one<T>(allocator, a_value);
183 }
184 } else {
185 if (b) {
186 if constexpr (HybridStorage::storage_category() == StorageCategory::MoveOnly) {
187 // Move from b into a.
188 auto* b_value = b->template down_cast<T>();
189 util::construct_at(util::addressof(a), util::move(*b_value));
190 util::destroy_at(b_value);
191 } else {
192 DI_ASSERT(!b);
193 }
194 } else {
195 // Just destroy a.
196 util::destroy_at(util::addressof(a));
197 }
198 }
199 }
