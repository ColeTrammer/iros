320 template<concepts::Serializable<Serializer> T, typename U = meta::Decay<T>,
321 typename MessageList = meta::MessageTypes<Type>>
322 requires(concepts::DecayConstructible<T> && meta::Contains<MessageList, U>)
323 friend auto tag_invoke(Tag<send>, Type self, T&& message, Optional<u32> maybe_message_number = {}) {
324 constexpr usize message_index = meta::Lookup<U, MessageList>;
325 static_assert(math::representable_as<u32>(message_index),
326 "There can be at most 2^32 messages in a protocol.");
327
328 // FIXME: acquire a async mutex first.
329 return just_from([message = util::forward<T>(message), data = self.data,
330 maybe_message_number] -> Result<u32> {
331 // Serialize the message. We re-use the send buffer to prevent extra allocations.
332 auto send_buffer = di::move(data->send_buffer);
333 send_buffer.clear();
334
335 auto total_size = sizeof(MessageHeader) + serialize_size(binary_format, message);
336 if constexpr (concepts::FallibleAllocator<Alloc>) {
337 DI_TRY(send_buffer.reserve(total_size));
338 } else {
339 send_buffer.reserve(total_size);
340 }
341
342 auto writer = VectorWriter(di::move(send_buffer));
343 auto message_number =
344 maybe_message_number.has_value() ? *maybe_message_number : data->message_number++;
345 auto message_header = MessageHeader {
346 .message_type = static_cast<u32>(message_index),
347 .message_number = message_number,
348 .message_size = total_size,
349 };
350 auto as_bytes = di::bit_cast<Array<byte, sizeof(MessageHeader)>>(message_header);
351 DI_TRY(write_exactly(writer, as_bytes.span()));
352 DI_TRY(serialize_binary(writer, message));
353 data->send_buffer = di::move(writer).vector();
354 return message_number;
355 }) |
356 let_value([data = self.data](u32 message_number) {
357 if constexpr (concepts::MessageWithReply<U>) {
358 return async_write_exactly(data->write, data->send_buffer.span()) |
359 let_value([data, message_number] {
360 return WaitForReplySender<Proto, Read, Write, Alloc, ClientOrServer,
361 message_index>(data, message_number);
362 });
363 } else {
364 return async_write_exactly(data->write, data->send_buffer.span());
365 }
366 });
367 }

369 ConnectionData<Proto, Read, Write, Alloc>* data;
370 };

405 template<typename Proto, typename Read, typename Write, typename TxFun, typename RxFun, typename Alloc,

407 struct MakeJoinedSender {
408 using Data = ConnectionData<Proto, Read, Write, Alloc>;
409 using Token = ConnectionToken<Proto, Read, Write, Alloc, ClientOrServer>;
410
411 static auto make_rx_sequence(Data* data, RxFun&& rx_function) {
412 return filter_messages<Proto, Read, Write, TxFun, RxFun, Alloc, ClientOrServer>(
413 message_sequence<Proto, Read, Write, Alloc, ClientOrServer>(data), data) |
414 let_value_each([data, rx_function = di::forward<RxFun>(rx_function)]<typename T>(
415 MessageHeader& header, T& message) mutable {
416 return get_env() | let_value([data, &header, rx_function = di::move(rx_function),
417 &message]<typename Env>(Env&) mutable {
418 auto call_function = [&] {
419 if constexpr (concepts::Invocable<RxFun&, T, Token>) {
420 return di::invoke(rx_function, di::move(message), Token(data));
421 } else if constexpr (concepts::Invocable<RxFun&, T>) {
422 return di::invoke(rx_function, di::move(message));
423 } else {
424 static_assert(di::concepts::AlwaysFalse<T>,
425 "Receiving function must accept all message types.");
426 }
427 };
428
429 auto sender_function = [&] {
430 using R = decltype(call_function());
431 if constexpr (!concepts::Sender<R>) {
432 if constexpr (concepts::LanguageVoid<R>) {
433 call_function();
434 return just();
435 } else {
436 return just(call_function());
437 }
438 } else {
439 return call_function();
440 }
441 };
442
443 using Send = decltype(sender_function());
444 static_assert(concepts::SingleSender<Send, Env>,
445 "Receiving function must return a single sender.");
446
447 using Value = meta::SingleSenderValueType<Send, Env>;
448 if constexpr (concepts::LanguageVoid<Value>) {
449 return sender_function();
450 } else if constexpr (concepts::MessageWithReply<T>) {
451 static_assert(
452 di::SameAs<meta::MessageReply<T>, Value>,
453 "Receiving function return a sender which sends the correct reply.");
454 return sender_function() | let_value([data, &header](Value& reply) {
455 return send(Token(data), di::move(reply), header.message_number);
456 });
457 } else if constexpr (meta::Contains<meta::MessageTypes<Token>, Value>) {
458 return sender_function() | let_value([data](Value& message) {
459 return send(Token(data), di::move(message));
460 });
461 } else {
462 static_assert(di::concepts::AlwaysFalse<T, Value>,
463 "Receiving function must return a sender which sends void or a "
464 "valid message type.");
465 }
466 });
467 });
468 }

470 static auto make_rx_sender(Data* data, RxFun&& rx_function) {
471 return ignore_all(make_rx_sequence(data, di::forward<RxFun>(rx_function))) | let_error([](auto&&) {
472 return just();
473 });
474 }

480 when_all(di::invoke(di::forward<TxFun>(tx_function), Token(data)),
481 make_rx_sender(data, di::forward<RxFun>(rx_function))));
