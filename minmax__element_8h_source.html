12 template<concepts::ForwardIterator It, concepts::SentinelFor<It> Sent, typename Proj = function::Identity,
13 concepts::IndirectStrictWeakOrder<meta::Projected<It, Proj>> Comp = function::Compare>
14 constexpr MinMaxResult<It> operator()(It first, Sent last, Comp comp = {}, Proj proj = {}) const {
15 if (first == last) {
16 return { first, first };
17 }
18
19 auto min_iter = first;
20 auto max_iter = first;
21 for (auto it = ++first; it != last; ++it) {
22 // Perform fancy algorithm to only do 3*N/2 - 2 comparisons, instead of 2*N comparisons.
23 // The idea is to consider pairs of elements in a row, and first compare them. Based on that
24 // result, we know whether to compare each element with the min or max element.
25
26 auto jt = container::next(it);
27 // Base case: only 1 element remains.
28 if (jt == last) {
29 auto compare_with_max =
30 function::invoke(comp, function::invoke(proj, *it), function::invoke(proj, *max_iter));
31 if (compare_with_max >= 0) {
32 max_iter = it;
33 } else if (function::invoke(comp, function::invoke(proj, *it), function::invoke(proj, *min_iter)) <
34 0) {
35 min_iter = it;
36 }
37 } else {
38 auto it_jt_result =
39 function::invoke(comp, function::invoke(proj, *it), function::invoke(proj, *jt));
40 if (it_jt_result < 0) {
41 // Compare it to min, jt to max.
42 if (function::invoke(comp, function::invoke(proj, *it), function::invoke(proj, *min_iter)) <
43 0) {
44 min_iter = it;
45 }
46 if (function::invoke(comp, function::invoke(proj, *jt), function::invoke(proj, *max_iter)) >=
47 0) {
48 max_iter = jt;
49 }
50 } else {
51 // Compare it to max, jt to min.
52 if (function::invoke(comp, function::invoke(proj, *it), function::invoke(proj, *min_iter)) >=
53 0) {
54 max_iter = it;
55 }
56 if (function::invoke(comp, function::invoke(proj, *jt), function::invoke(proj, *max_iter)) <
57 0) {
58 min_iter = jt;
59 }
60 }
61 ++it;
62 }
63 }
64 return { util::move(min_iter), util::move(max_iter) };
65 }
